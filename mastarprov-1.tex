\section{Hantverkare}

Algoriritmen använder använder sig av en vektor innehållandes tiderna för de $n$ st olika hantverkarna.

\begin{algorithm}
\begin{algorithmic}[0]
\State $cost \gets$ 0, $times \gets [t_1,...,t_n]$
\While{$times$ is not empty}
\State  \textbf{extract} lowest value from $times$  \textbf{as} $t$
\State $cost = t * |times|$
\EndWhile
\State  \textbf{return} $i$
\end{algorithmic}
\end{algorithm}
